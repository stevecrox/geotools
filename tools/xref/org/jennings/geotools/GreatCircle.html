54      /**
55       *
56       * Given two geographic points find the distance and bearing between the
57       * points.
58       *
59       * @param lon1
60       * @param lat1
61       * @param lon2
62       * @param lat2
63       * @return
64       */
65      public DistanceBearing getDistanceBearing(double lon1, double lat1, double lon2, double lat2) {
66  
67          double gcDist;
68          double bearing;
69  
70          double lon1R = lon1 * CONSTANTS.D2R;
71          double lat1R = lat1 * CONSTANTS.D2R;
72          double lon2R = lon2 * CONSTANTS.D2R;
73          double lat2R = lat2 * CONSTANTS.D2R;
74  
75          /*
76              Functions are a little whacky around the north and south pole.
77              The only valid bearing from north pole is -180.
78              I wouldn't trust the program for points near the poles.
79           */
80          if (Math.abs(lat1 - 90.0) < 0.00001) {
81              // very close to north pole distance R * theta            
82              double l = 90.0 - lat2;
83              gcDist = CONSTANTS.EARTH_RADIUS * l * CONSTANTS.D2R / 1000.0;
84              // let bearing in lon2
85              bearing = lon2;
86          } else if (Math.abs(lat1 + 90) < 0.00001) {
87              // very close to south pole distance R * theta
88              double l = lat2 + 90.0;
89              gcDist = CONSTANTS.EARTH_RADIUS * l * CONSTANTS.D2R / 1000.0;
90              bearing = lon2;
91  
92          } else {
93  
94              // law of Cosines
95              double lambda = Math.abs(lon2R - lon1R);
96  
97              double x1 = Math.cos(lat2R) * Math.sin(lambda);
98  
99              double x2 = Math.cos(lat1R) * Math.sin(lat2R)
100                     - Math.sin(lat1R) * Math.cos(lat2R) * Math.cos(lambda);
101 
102             double x3 = Math.sin(lat1R) * Math.sin(lat2R)
103                     + Math.cos(lat1R) * Math.cos(lat2R) * Math.cos(lambda);
104 
105             double x4 = Math.sqrt(x1 * x1 + x2 * x2);
106 
107             double sigma = Math.atan2(x4, x3);
108 
109             gcDist = sigma * CONSTANTS.EARTH_RADIUS / 1000.0;
110 
111             double y1 = Math.sin(lon2R - lon1R) * Math.cos(lat2R);
112 
113             double y2 = Math.cos(lat1R) * Math.sin(lat2R)
114                     - Math.sin(lat1R) * Math.cos(lat2R) * Math.cos(lon2R - lon1R);
115 
116             double y3 = Math.atan2(y1, y2);
117 
118             bearing = (y3 * CONSTANTS.R2D) % 360;
119 
120             // Convert to value from -180 to 180
121             bearing = bearing > 180.0 ? bearing - 360.0 : bearing;
122 
123         }
124 
125         DistanceBearing distB = new DistanceBearing(gcDist, bearing);
126 
127         return distB;
128     }

130     /**
131      *
132      * Given a point and a distance and bearing find a new point.
133      *
134      * @param lon1
135      * @param lat1
136      * @param distance
137      * @param bearing
138      * @return
139      */
140     public GeographicCoordinate getNewCoordPair(double lon1, double lat1, double distance, double bearing) {
141         DistanceBearing distB = new DistanceBearing(distance, bearing);
142         GeographicCoordinate coord1 = new GeographicCoordinate(lon1, lat1);
143         return getNewCoordPair(coord1, distB);
144     }

146     /**
147      *
148      * @param coord1
149      * @param distB
150      * @return
151      */
152     public GeographicCoordinate getNewCoordPair(GeographicCoordinate coord1, DistanceBearing distB) {
153 
154         double lat1 = coord1.getLat();
155         double lon1 = coord1.getLon();
156         double lat2 = 0.0;
157         double lon2 = 0.0;
158 
159         boolean bln360 = false;
160 
161         // Allow for lon values 180 to 360 (adjust them to -180 to 0)
162         double lonDD = lon1;
163         if (lonDD > 180.0 && lonDD <= 360) {
164             lonDD = lonDD - 360;
165             lon1 = lonDD;
166             bln360 = true;
167         }
168 
169         double alpha;
170         double l;
171         double k;
172         double gamma;
173         double phi;
174         //double theta;
175         //double hdng2;
176 
177         double hdng = distB.getBearing();
178 
179         if (hdng < 0) {
180             hdng = hdng + 360;
181         }
182 
183         // Round the input            
184         BigDecimal bd = new BigDecimal(hdng);
185         bd = bd.setScale(6, BigDecimal.ROUND_HALF_UP);
186         hdng = bd.doubleValue();
187 
188         double dist = distB.getDistance() * 1000;
189 
190         if (lat1 == 90 || lat1 == -90) {
191             // hdng doesn't make a lot of since at the poles assume this is just the lon
192             lon2 = hdng;
193             alpha = dist / CONSTANTS.EARTH_RADIUS;
194             if (lat1 == 90) {
195                 lat2 = 90 - alpha * CONSTANTS.R2D;
196             } else {
197                 lat2 = -90 + alpha * CONSTANTS.R2D;
198             }
199 
200         } else if (hdng == 0 || hdng == 360) {
201             // going due north within some rounded number
202             alpha = dist / CONSTANTS.EARTH_RADIUS;
203             lat2 = lat1 + alpha * CONSTANTS.R2D;
204             lon2 = lon1;
205         } else if (hdng == 180) {
206             // going due south witin some rounded number
207             alpha = dist / CONSTANTS.EARTH_RADIUS;
208             lat2 = lat1 - alpha * CONSTANTS.R2D;
209             lon2 = lon1;
210         } else if (hdng == 90) {
211             lat2 = lat1;
212             l = 90 - lat1;
213             alpha = dist / CONSTANTS.EARTH_RADIUS / Math.sin(l * CONSTANTS.D2R);
214             //phi = Math.asin(Math.sin(alpha)/ Math.sin(l*D2R));                 
215             lon2 = lon1 + alpha * CONSTANTS.R2D;
216         } else if (hdng == 270) {
217             lat2 = lat1;
218             l = 90 - lat1;
219             alpha = dist / CONSTANTS.EARTH_RADIUS / Math.sin(l * CONSTANTS.D2R);
220             //phi = Math.asin(Math.sin(alpha)/ Math.sin(l*D2R));                       
221             lon2 = lon1 - alpha * CONSTANTS.R2D;
222         } else if (hdng > 0 && hdng < 180) {
223             l = 90 - lat1;
224             alpha = dist / CONSTANTS.EARTH_RADIUS;
225             k = Math.acos(Math.cos(alpha) * Math.cos(l * CONSTANTS.D2R)
226                     + Math.sin(alpha) * Math.sin(l * CONSTANTS.D2R) * Math.cos(hdng * CONSTANTS.D2R));
227             lat2 = 90 - k * CONSTANTS.R2D;
228             double n = (Math.cos(alpha) - Math.cos(k) * Math.cos(l * CONSTANTS.D2R))
229                     / (Math.sin(k) * Math.sin(l * CONSTANTS.D2R));
230             if (n > 1) {
231                 n = 1;
232             }
233             if (n < -1) {
234                 n = -1;
235             }
236             phi = Math.acos(n);
237             lon2 = lon1 + phi * CONSTANTS.R2D;
238             //theta = Math.sin(phi) * Math.sin(l * CONSTANTS.D2R) / Math.sin(alpha);
239             //hdng2 = 180 - theta * CONSTANTS.R2D;
240         } else if (hdng > 180 && hdng < 360) {
241             gamma = 360 - hdng;
242             l = 90 - lat1;
243             alpha = dist / CONSTANTS.EARTH_RADIUS;
244             k = Math.acos(Math.cos(alpha) * Math.cos(l * CONSTANTS.D2R)
245                     + Math.sin(alpha) * Math.sin(l * CONSTANTS.D2R) * Math.cos(gamma * CONSTANTS.D2R));
246             lat2 = 90 - k * CONSTANTS.R2D;
247             double n = (Math.cos(alpha) - Math.cos(k) * Math.cos(l * CONSTANTS.D2R))
248                     / (Math.sin(k) * Math.sin(l * CONSTANTS.D2R));
249             if (n > 1) {
250                 n = 1;
251             }
252             if (n < -1) {
253                 n = -1;
254             }
255             phi = Math.acos(n);
256             lon2 = lon1 - phi * CONSTANTS.R2D;
257             //theta = Math.sin(phi) * Math.sin(l * CONSTANTS.D2R) / Math.sin(alpha);
258             //hdng2 = 180 - theta * CONSTANTS.R2D;
259         }
260 
261         int decimalPlaces = 12;
262         bd = new BigDecimal(lat2);
263         bd = bd.setScale(decimalPlaces, BigDecimal.ROUND_HALF_UP);
264         lat2 = bd.doubleValue();
265 
266         bd = new BigDecimal(lon2);
267         bd = bd.setScale(decimalPlaces, BigDecimal.ROUND_HALF_UP);
268         lon2 = bd.doubleValue();
269 
270         if (lat2 > 90) {
271             lat2 = 180 - lat2;
272             lon2 = (lon2 + 180) % 360;
273         }
274 
275         if (lon2 > 180) {
276             lon2 = lon2 - 360;
277         }
278 
279         if (lat2 < -90) {
280             lat2 = -180 - lat2;
281             lon2 = (lon2 - 180) % 360;
282         }
283         if (lon2 < -180) {
284             lon2 = lon2 + 360;
285         }
286 
287         // adjust the lon back to 360 scale if input was like that
288         if (bln360) {
289             if (lon2 < 0) {
290                 lon2 = lon2 + 360;
291             }
292         }
293 
294         if (lon2 < -180.0) {
295             System.out.println(lon2);
296             System.out.println(distB.getBearing());
297 
298         }
299 
300         GeographicCoordinate nc = new GeographicCoordinate(lon2, lat2);
301 
302         return nc;
303     }

305     /**
306      *
307      * Given two points on a line find the slope; then given a value x return y
308      *
309      * @param x1
310      * @param y1
311      * @param x2
312      * @param y2
313      * @param x
314      * @return
315      */
316     public double findCrossing(double x1, double y1, double x2, double y2, double x) {
317         double m = (y2 - y1) / (x2 - x1);
318 
319         return m * (x - x1) + y1;
320     }

322     public static void main(String[] args) {
323 
324         // Example Command Line args: localhost 5565 faa-stream.csv 1000 10000
325         int numargs = args.length;
326         if (numargs != 5) {
327             // append append time option was added to support end-to-end latency; I used it for Trinity testing
328             System.err.println("Usage: GreatCircle newpt <lon> <lat> <dist> <bearing>");
329             System.err.println("Usage: GreatCircle distb <lon1> <lat1> <lon2> <lat2>");
330         } else {
331 
332             GreatCircle gc = new GreatCircle();
333             String req = args[0];
334 
335             if (req.equalsIgnoreCase("newpt")) {
336                 double lon = Double.parseDouble(args[1]);
337                 double lat = Double.parseDouble(args[2]);
338                 double dist = Double.parseDouble(args[3]);
339                 double bear = Double.parseDouble(args[4]);
340 
341                 GeographicCoordinate pt = gc.getNewCoordPair(lon, lat, dist, bear);
342                 System.out.println(String.format("%.5f", pt.getLon()) + "," + String.format("%.5f", pt.getLat()));
343 
344             } else if (req.equalsIgnoreCase("distb")) {
345                 double lon1 = Double.parseDouble(args[1]);
346                 double lat1 = Double.parseDouble(args[2]);
347                 double lon2 = Double.parseDouble(args[3]);
348                 double lat2 = Double.parseDouble(args[4]);
349 
350                 GeographicCoordinate pt1 = new GeographicCoordinate(lon1, lat1);
351                 GeographicCoordinate pt2 = new GeographicCoordinate(lon2, lat2);
352 
353                 DistanceBearing db = gc.getDistanceBearing(pt1, pt2);
354                 System.out.println(String.format("%.0f", db.getDistance()) + "," + String.format("%.2f", db.getBearing()));
355 
356             } else {
357                 System.out.println("First parameter must be newpt or distb");
358             }
359 
360         }
361     }
