35      /**
36       *
37       * Will not work for Ellipses where semi-major axis is greater than half
38       * circumference of Earth
39       *
40       * Also may have issues when ellipses extend into polar regions
41       *
42       * Uses GreatCircle to calculate points on the ellipse.
43       *
44       * @param clon Center Longitude
45       * @param clat Center Latitude
46       * @param a Semi-Major Axis (km)
47       * @param b Semi-Minor Axis (km)
48       * @param rotationDeg Clockwise rotation in degrees. For 0 semi-major axis
49       * is North-South
50       * @param numPoints The actual number of points returned will be
51       * approximately this amount
52       * @param isClockwise For GeoJSON set false; for Esri Geometries set true
53       * @return JSONArray with ellipse points
54       */
55      public JSONArray createEllipse(double clon, double clat, Double a, Double b, Double rotationDeg, Integer numPoints, boolean isClockwise) {
56  
57          GreatCircle gc = new GreatCircle();
58  
59          if (numPoints == null) {
60              numPoints = 20;
61          }
62  
63          if (a == null) {
64              // default to 500 meters or 0.50 km
65              a = 0.50;
66          }
67  
68          if (b == null) {
69              // default to 50 meters or 0.050 km
70              b = 0.050;
71          }
72  
73          if (rotationDeg == null) {
74              rotationDeg = 0.0;
75          }
76  
77          if (a < b) {
78              // Switch a and b
79              // Semimajor axis should be the larger of the two
80              double t = a;
81              a = b;
82              b = t;
83          }
84  
85          // If this was a circle then even distribuation would be d
86          double d = 360.0 / (numPoints - 1);
87  
88          // The number of points returns will depend on eccentricity
89          ArrayList<GeographicCoordinate> coords = new ArrayList<>();
90  
91          double v = 360.0;
92          GeographicCoordinate coord1 = new GeographicCoordinate(clon, clat);
93  
94          GeographicCoordinate lastcoord = coord1;
95  
96          while (v > 0.0001) {
97  
98              double r = v * CONSTANTS.D2R;
99              double sv = Math.sin(r);
100             double cv = Math.cos(r);
101             double radiusKM = a * b / Math.sqrt(a * a * sv * sv + b * b * cv * cv);
102 
103             double bearing = v;
104             if (bearing > 180) {
105                 bearing = bearing - 360;
106             }
107             DistanceBearing distb = new DistanceBearing(radiusKM, bearing);
108 
109             if (rotationDeg != 0.0) {
110                 bearing = (distb.getBearing() + rotationDeg) % 360.0;
111                 if (bearing > 180) {
112                     bearing = bearing - 360;
113                 }
114                 distb.setBearing(bearing);
115             }
116 
117             GeographicCoordinate nc = gc.getNewCoordPair(coord1, distb);
118 
119             // Only add the nc if it is different from previous
120             double lon = nc.getLon();
121             double lat = nc.getLat();
122 
123             if (Math.abs(lon - lastcoord.getLon()) < 0.00001 && Math.abs(lat - lastcoord.getLat()) < 0.00001) {
124                 // skip this point too close to previous
125             } else {
126                 coords.add(nc);
127                 if (debug) {
128                     System.out.println(v + "," + bearing + "," + radiusKM + "," + lon + "," + lat);
129                 }
130                 lastcoord = nc;
131             }
132 
133             // Densifying the points around 0 and 360 and 180; just 10 degrees out really improved shape of ellipse 
134             if (v > 350 || v < 10 || (v > 170 && v < 190)) {
135                 v = v - d * b / a;
136             } else {
137                 v = v - d;
138             }
139 
140         }
141 
142         // Added code to prevent adding a last point that was too close to the end point.
143         GeographicCoordinate firstPoint = coords.get(0);
144         GeographicCoordinate lastPoint = coords.get(coords.size() - 1);
145 
146         if (Math.abs(firstPoint.getLon() - lastPoint.getLon()) < 0.00001 && Math.abs(firstPoint.getLat() - lastPoint.getLat()) < 0.00001) {
147             // Replace last point with first
148             coords.set(coords.size() - 1, firstPoint);
149         } else {
150             // Append firstpoint 
151             coords.add(firstPoint);
152             if (debug) {
153                 System.out.println(-1 + "," + -1 + "," + -1 + "," + firstPoint.getLon() + "," + firstPoint.getLat());
154             }
155 
156         }
157 
158 //        System.exit(0);
159         GeographicCoordinate nc1;
160         GeographicCoordinate nc2;
161 
162         JSONArray[] exteriorRing = new JSONArray[2];
163         exteriorRing[0] = new JSONArray();
164         exteriorRing[1] = new JSONArray();
165 
166         int ringNum = 0;
167 
168         nc1 = coords.get(0);
169         int i = 0;
170         double lon1 = nc1.getLon();
171         double lat1 = nc1.getLat();
172         if (debug) {
173             System.out.println(i + ":" + lon1 + "," + lat1);
174         }
175         JSONArray coord = new JSONArray("[" + lon1 + ", " + lat1 + "]");
176         exteriorRing[ringNum].put(coord);
177         i++;
178 
179         boolean crossedDTEast = false;
180         boolean crossedDTWest = false;
181         boolean crossedZeroEast = false;
182         boolean crossedZeroWest = false;
183 
184         while (i < coords.size()) {
185 
186             nc2 = coords.get(i);
187             // Compare coordinates
188             double lon2 = nc2.getLon();
189             double lat2 = nc2.getLat();
190 
191             if (lon1 == 0.0 && lon2 < 0.0) {
192                 crossedZeroWest = true;
193 
194                 // Switch ringnum
195                 if (ringNum == 0) {
196                     ringNum = 1;
197                 } else if (ringNum == 1) {
198                     ringNum = 0;
199                 }
200 
201                 // Add nc1 to ring
202                 coord = new JSONArray("[" + lon1 + ", " + lat1 + "]");
203                 exteriorRing[ringNum].put(coord);
204 
205             } else if (lon2 == 0.0 && lon1 < 0.0) {
206 
207                 crossedZeroEast = true;
208 
209                 // Add nc2 to ring
210                 coord = new JSONArray("[" + lon2 + ", " + lat2 + "]");
211                 exteriorRing[ringNum].put(coord);
212 
213                 if (crossedZeroWest) {
214                     // close ring
215                     exteriorRing[ringNum].put(exteriorRing[ringNum].get(0));
216                 }
217 
218                 // Switch ringnum
219                 if (ringNum == 0) {
220                     ringNum = 1;
221                 } else if (ringNum == 1) {
222                     ringNum = 0;
223                 }
224 
225             } else if (lon2 > 0.0 && lon1 < 0.0) {
226 
227                 if (Math.abs(lon2 - lon1) > 180.0) {
228                     // Crossing DT heading west
229                     if (debug) {
230                         System.out.println("Crossing DT heading west");
231                     }
232                     crossedDTWest = true;
233 
234                     double x1 = (lon1 < 0) ? lon1 + 360.0 : lon1;
235                     double x2 = (lon2 < 0) ? lon2 + 360.0 : lon2;
236                     double crossing = gc.findCrossing(x1, lat1, x2, lat2, 180.0);
237                     if (debug) {
238                         System.out.println(crossing);
239                     }
240 
241                     // Add point to current ring
242                     coord = new JSONArray("[-180.0, " + crossing + "]");
243                     exteriorRing[ringNum].put(coord);
244 
245                     if (crossedZeroWest) {
246                         // South Pole in Footprint
247                         coord = new JSONArray("[-180.0, -90.0]");
248                         exteriorRing[ringNum].put(coord);
249                         coord = new JSONArray("[0.0, -90.0]");
250                         exteriorRing[ringNum].put(coord);
251                     }
252 
253                     // Swith to other ring
254                     if (ringNum == 0) {
255                         ringNum = 1;
256                     } else if (ringNum == 1) {
257                         ringNum = 0;
258                     }
259 
260                     if (crossedZeroWest) {
261                         // South Pole in Footprint
262                         coord = new JSONArray("[0.0, -90.0]");
263                         exteriorRing[ringNum].put(coord);
264                         coord = new JSONArray("[180, -90.0]");
265                         exteriorRing[ringNum].put(coord);
266                     }
267 
268                     // Add point to other ring
269                     coord = new JSONArray("[180.0, " + crossing + "]");
270                     exteriorRing[ringNum].put(coord);
271 
272                 } else {
273                     // or gone from -1.xx to 1.xx  (Crossing 0 heading east)
274                     if (debug) {
275                         System.out.println("Crossing 0 heading east");
276                         System.out.println("lon1 = " + lon1);
277                         System.out.println("lon2 = " + lon2);
278                     }
279 
280                     crossedZeroEast = true;
281 
282                     // -1.xx to 1.00
283                     double x1 = lon1;
284                     double x2 = lon2;
285                     double crossing = gc.findCrossing(x1, lat1, x2, lat2, 0.0);
286                     if (debug) {
287                         System.out.println(crossing);
288                     }
289 
290                     // Add point to current ring
291                     coord = new JSONArray("[0.0, " + crossing + "]");
292                     exteriorRing[ringNum].put(coord);
293 
294                     if (crossedDTEast) {
295                         // North Pole is in footprint add 180,90 and 0, 90 
296                         coord = new JSONArray("[0.0,90.0]");
297                         exteriorRing[ringNum].put(coord);
298                         coord = new JSONArray("[-180.0,90.0]");
299                         exteriorRing[ringNum].put(coord);
300                     }
301 
302                     // Swith to other ring
303                     if (ringNum == 0) {
304                         ringNum = 1;
305                     } else if (ringNum == 1) {
306                         ringNum = 0;
307                     }
308 
309                     if (crossedDTEast) {
310                         // North pole is in foot print add point 0,90, -180,90  
311                         coord = new JSONArray("[180.0,90.0]");
312                         exteriorRing[ringNum].put(coord);
313                         coord = new JSONArray("[0.0,90.0]");
314                         exteriorRing[ringNum].put(coord);
315 
316                     }
317 
318                     // Add point to other ring
319                     coord = new JSONArray("[0.0, " + crossing + "]");
320                     exteriorRing[ringNum].put(coord);
321 
322                 }
323 
324             } else if (lon2 < 0.0 && lon1 >= 0.0) {
325 
326                 if (Math.abs(lon2 - lon1) > 180.0) {
327                     // Either gone from 179.xx to -179.xx (Crossing DT heading west)                    
328                     if (debug) {
329                         System.out.println("Crossing DT heading east");
330                     }
331                     crossedDTEast = true;
332 
333                     double x1 = (lon1 < 0) ? lon1 + 360.0 : lon1;
334                     double x2 = (lon2 < 0) ? lon2 + 360.0 : lon2;
335                     double crossing = gc.findCrossing(x1, lat1, x2, lat2, 180.0);
336                     if (debug) {
337                         System.out.println(crossing);
338                     }
339 
340                     coord = new JSONArray("[180.0, " + crossing + "]");
341                     exteriorRing[ringNum].put(coord);
342 
343                     if (crossedZeroEast) {
344                         // North Pole in Footprint
345                         coord = new JSONArray("[180.0, 90.0]");
346                         exteriorRing[ringNum].put(coord);
347                         coord = new JSONArray("[0.0, 90.0]");
348                         exteriorRing[ringNum].put(coord);
349                     }
350 
351                     // Swith to other ring
352                     if (ringNum == 0) {
353                         ringNum = 1;
354                     } else if (ringNum == 1) {
355                         ringNum = 0;
356                     }
357 
358                     if (crossedZeroEast) {
359                         // North Pole in Footprint
360                         coord = new JSONArray("[0.0, 90.0]");
361                         exteriorRing[ringNum].put(coord);
362                         coord = new JSONArray("[-180, 90.0]");
363                         exteriorRing[ringNum].put(coord);
364                     }
365 
366                     coord = new JSONArray("[-180.0, " + crossing + "]");
367                     exteriorRing[ringNum].put(coord);
368 
369                 } else {
370                     if (debug) {
371                         System.out.println("Crossing 0 heading west");
372                     }
373                     crossedZeroWest = true;
374 
375                     // or gone from 1.xx to -1.xx (Crossing 0 heading west 
376                     double x1 = lon1;
377                     double x2 = lon2;
378                     double crossing = gc.findCrossing(x1, lat1, x2, lat2, 0.0);
379                     if (debug) {
380                         System.out.println(crossing);
381                     }
382 
383                     // Add point to current ring
384                     coord = new JSONArray("[0.0, " + crossing + "]");
385                     exteriorRing[ringNum].put(coord);
386 
387                     if (crossedDTWest) {
388                         // South Pole is in footprint 
389                         coord = new JSONArray("[0.0,-90.0]");
390                         exteriorRing[ringNum].put(coord);
391                         coord = new JSONArray("[180.0,-90.0]");
392                         exteriorRing[ringNum].put(coord);
393                     }
394 
395                     // Swith to other ring
396                     if (ringNum == 0) {
397                         ringNum = 1;
398                     } else if (ringNum == 1) {
399                         ringNum = 0;
400                     }
401 
402                     if (crossedDTWest) {
403                         // North pole is in foot print add point 0,90, -180,90  
404                         coord = new JSONArray("[-180.0,-90.0]");
405                         exteriorRing[ringNum].put(coord);
406                         coord = new JSONArray("[0.0,-90.0]");
407                         exteriorRing[ringNum].put(coord);
408 
409                     }
410 
411                     // Add point to other ring
412                     coord = new JSONArray("[0.0, " + crossing + "]");
413                     exteriorRing[ringNum].put(coord);
414                 }
415             } else if (lon1 == 0.0) {
416                 if (lon2 > 0.0) {
417 
418                 } else {
419 
420                 }
421             }
422 
423             if (debug) {
424                 System.out.println(i + ":" + lon2 + "," + lat2);
425             }
426 
427             if (i < coords.size() - 1) {
428                 // Inject last coord
429                 coord = new JSONArray("[" + lon2 + ", " + lat2 + "]");
430                 exteriorRing[ringNum].put(coord);
431             }
432 
433             nc1 = nc2;
434             lon1 = nc1.getLon();
435             lat1 = nc1.getLat();
436             i++;
437 
438         }
439 
440         JSONArray polys = new JSONArray();
441         JSONArray poly;
442 
443 
444         for (i = 0; i < 2; i++) {
445             poly = new JSONArray();
446             if (exteriorRing[i].length() > 0) {
447                 // Add last point same as first
448                 exteriorRing[i].put(exteriorRing[i].get(0));
449 
450                 if (isClockwise) {
451                     // Reverse order 
452                     JSONArray reversedPoly = new JSONArray();
453                     int j = exteriorRing[i].length() - 1;
454                     while (j >= 0) {
455                         reversedPoly.put(exteriorRing[i].get(j));
456                         j--;
457                     }
458                     polys.put(reversedPoly);
459                 } else {
460                     poly.put(exteriorRing[i]);
461                     polys.put(poly);
462                 }
463                 
464             }
465         }
466 
467         if (debug) {
468             int j = 0;
469             while (j < 2) {
470                 System.out.println("exteriorRing[" + j + "]");
471                 i = 0;
472 
473                 while (i < exteriorRing[j].length()) {
474                     System.out.println(exteriorRing[j].get(i));
475                     i++;
476                 }
477                 j++;
478             }
479         }
480         // Create the Geom
481         return polys;
482 
483     }

485     public String createEsrijsonEllipse(int id, double lon, double lat, double a, double b, double rot, int numPoints) {
486 
487         Ellipse el = new Ellipse();
488 
489         // Clockwise Poly
490         JSONArray polys;
491         polys = el.createEllipse(lon, lat, a, b, rot, numPoints, true);
492 
493         JSONArray rngs = new JSONArray();
494         rngs.put(polys.getJSONArray(0).getJSONArray(0));
495         try {
496             rngs.put(polys.getJSONArray(1).getJSONArray(0));
497         } catch (Exception e) {
498             // ok to ignore
499         }
500         JSONObject geom = new JSONObject();
501         geom.put("rings", rngs);
502 
503         JSONObject esriJson = new JSONObject();
504         esriJson.put("a", a);
505         esriJson.put("b", b);
506         esriJson.put("rot", rot);
507         esriJson.put("clon", lon);
508         esriJson.put("clat", lat);
509         esriJson.put("geometry", geom);
510 
511         return esriJson.toString();
512     }

514     public String createWktEllipse(double lon, double lat, double a, double b, double rot, int numPoints) {
515         
516         
517         JSONArray polys = createEllipse(lon, lat, a, b, rot, numPoints, false);
518         
519         int numPolys = polys.length();
520         
521         StringBuffer buf = new StringBuffer();
522         
523         if (numPolys == 1) {
524             // Polygon
525             buf.append("POLYGON ((");
526             JSONArray ring = polys.getJSONArray(0).getJSONArray(0);
527             int numPts = ring.length();
528             //System.out.println("numPts:" + numPts);
529             int cnt = 0;
530             while (cnt < numPts) {
531                 JSONArray pt = ring.getJSONArray(cnt);
532                 //System.out.println(pt);
533                 buf.append(String.valueOf(pt.getDouble(0)) + " " + String.valueOf(pt.getDouble(1)));
534                 cnt++;
535                 if (cnt < numPts) {
536                     buf.append(",");
537                 }                
538             }            
539             buf.append("))");
540         } else {
541             // Multipolygon
542             buf.append("MULTIPOLYGON (");
543             
544             int cntPoly = 0;
545             while (cntPoly < numPolys) {
546                 JSONArray ring = polys.getJSONArray(cntPoly).getJSONArray(0);
547                 int numPts = ring.length();                
548                 int cntPt = 0;
549                 buf.append("((");
550                 while (cntPt < numPts) {
551                     JSONArray pt = ring.getJSONArray(cntPt);
552                     //System.out.println(pt);
553                     buf.append(String.valueOf(pt.getDouble(0)) + " " + String.valueOf(pt.getDouble(1)));
554                     cntPt++;
555                     if (cntPt < numPts) {
556                         buf.append(",");
557                     }                
558                 }
559                 buf.append("))");
560                 cntPoly++;
561                 if (cntPoly < numPolys) {
562                     buf.append(",");
563                 }                               
564             }
565             
566             buf.append(")");
567         }
568         System.out.println(numPolys);
569         
570         
571         
572         
573         return buf.toString();
574         
575     }

578     public String createGeoJsonEllipse(int id, double lon, double lat, double a, double b, double rot, int numPoints) {
579 
580 
581         JSONObject featureCollection = new JSONObject();
582         featureCollection.put("type", "FeatureCollection");
583 
584         JSONArray features = new JSONArray();
585 
586         JSONObject feature = new JSONObject();
587         feature.put("type", "Feature");
588 
589         // Add properties (At least one is required)
590         JSONObject properties = new JSONObject();
591         properties.put("a", a);
592         properties.put("b", b);
593         properties.put("rot", rot);
594         properties.put("clon", lon);
595         properties.put("clat", lat);
596         feature.put("properties", properties);
597 
598         // Get the coordinates
599         JSONArray coords = createEllipse(lon, lat, a, b, rot, numPoints, false);
600 
601         JSONObject geom = new JSONObject();
602         geom.put("type", "MultiPolygon");
603         geom.put("coordinates", coords);
604 
605         feature.put("geometry", geom);
606 
607         features.put(feature);
608         featureCollection.put("features", features);
609 
610         return featureCollection.toString();
611     }

615     public String createEllipse(int id, double lon, double lat, double a, double b, double rot, int numPoints, GeomType geomType) {
616         switch (geomType) {
617             case EsriJson:
618                 return createEllipse(lon, lat, a, b, rot, numPoints, true).toString();
619             case GeoJson:      
620                 return createEllipse(lon, lat, a, b, rot, numPoints, false).toString();
621             case Wkt:      
622                 return createWktEllipse(lon, lat, a, b, rot, numPoints);
623         }
624         return null;
625         
626     }
